/// 简介

引擎的开发工具决定了团队的迭代速度。一个好用的关卡编辑器、实时性能分析器、内存调试器能让开发效率提升数倍——"调试 10 分钟解决的问题，没有工具可能要盲猜两天。"

本章覆盖游戏引擎常见的内建工具：关卡编辑器设计、ImGui 即时模式 GUI、性能分析（Profiling）、内存调试、资产管道工具、日志系统和命令行参数处理。重点是 **设计思路和架构**，而非特定 API 的教程。

// 关卡编辑器设计

关卡编辑器是引擎最复杂的工具——它本身就是一个完整的应用程序。

\diagram[flow]{
编辑器(Editor) -> 场景数据(.scene JSON) -> 引擎运行时(Runtime) -> 游戏画面
}

\table{
编辑器功能 | 实现要点
---
场景视口 | 渲染引擎嵌入到编辑器窗口，支持自由摄像机
对象选取 | 鼠标射线 (Ray Picking) 或颜色 ID 缓冲法
属性面板 | 反射系统 + ImGui，自动为组件生成编辑控件
Undo/Redo | 命令模式：每个操作封装为 Command 对象，压入撤销栈
Gizmo | 平移/旋转/缩放 3D 手柄，ImGuizmo 库
保存/加载 | 将场景序列化为 JSON/二进制，版本化格式
运行模式 | "Play"时序列化当前场景 → 运行 → "Stop"时恢复序列化快照
}

\code[cpp]{
// Undo/Redo 系统核心
class ICommand {
public:
    virtual ~ICommand() = default;
    virtual void execute() = 0;
    virtual void undo() = 0;
    virtual std::string description() const = 0;
};

class CommandHistory {
    std::vector<std::unique_ptr<ICommand>> undoStack;
    std::vector<std::unique_ptr<ICommand>> redoStack;
public:
    void execute(std::unique_ptr<ICommand> cmd) {
        cmd->execute();
        undoStack.push_back(std::move(cmd));
        redoStack.clear(); // 新操作清除 redo 历史
    }
    void undo() {
        if (undoStack.empty()) return;
        undoStack.back()->undo();
        redoStack.push_back(std::move(undoStack.back()));
        undoStack.pop_back();
    }
    void redo() {
        if (redoStack.empty()) return;
        redoStack.back()->execute();
        undoStack.push_back(std::move(redoStack.back()));
        redoStack.pop_back();
    }
};
}

// Dear ImGui 即时模式 GUI

**Dear ImGui** 是游戏引擎开发者最常用的调试 UI 库。**即时模式（Immediate Mode）** 意味着每帧重建 UI——没有持久的 Widget 对象，不需要维护 UI 状态树。

\code[cpp]{
#include <imgui.h>

void renderDebugUI(GameState& state, float dt) {
    // 性能面板
    ImGui::Begin("Performance");
    ImGui::Text("FPS: %.1f", 1.0f / dt);
    ImGui::Text("Frame: %.2f ms", dt * 1000.0f);
    ImGui::PlotLines("FPS", fpsHistory.data(), fpsHistory.size());
    ImGui::End();

    // 场景检查器
    ImGui::Begin("Inspector");
    for (auto& obj : state.scene.objects) {
        if (ImGui::TreeNode(obj.name.c_str())) {
            ImGui::DragFloat3("Position", &obj.position.x, 0.1f);
            ImGui::DragFloat("Rotation", &obj.rotation, 1.0f);
            ImGui::DragFloat3("Scale", &obj.scale.x, 0.01f);
            if (ImGui::Button("Delete")) state.pendingDelete.push_back(obj.id);
            ImGui::TreePop();
        }
    }
    ImGui::End();

    // 引擎控制台
    ImGui::Begin("Console");
    for (auto& line : console.log)
        ImGui::TextUnformatted(line.c_str());
    static char inputBuf[256] = "";
    if (ImGui::InputText("##cmd", inputBuf, sizeof(inputBuf),
                         ImGuiInputTextFlags_EnterReturnsTrue)) {
        console.execute(inputBuf);
        inputBuf[0] = '\0';
    }
    ImGui::End();
}
}

\tip{**ImGui 是学习引擎开发的最佳伴侣。** 几行代码就能创建滑块、颜色选择器、曲线编辑器。用它实时调整物理参数、着色器变量、AI 权重——比改代码重编译快百倍。}

// 性能分析——帧分析器

\code[cpp]{
class FrameProfiler {
    struct ScopeRecord {
        std::string name;
        float startMs;
        float endMs;
        int depth;
    };
    std::vector<ScopeRecord> currentFrame;
    std::vector<ScopeRecord> lastFrame; // 供 UI 读取
    int currentDepth = 0;

public:
    struct ScopeGuard {
        FrameProfiler& profiler;
        ScopeGuard(FrameProfiler& p, std::string_view name) : profiler(p) {
            p.beginScope(name);
        }
        ~ScopeGuard() { profiler.endScope(); }
    };

    void beginFrame() {
        lastFrame = std::move(currentFrame);
        currentFrame.clear();
        currentDepth = 0;
    }

    void beginScope(std::string_view name) {
        currentFrame.push_back({std::string(name), getTimeMs(), 0, currentDepth++});
    }
    void endScope() {
        for (auto it = currentFrame.rbegin(); it != currentFrame.rend(); ++it) {
            if (it->endMs == 0) { it->endMs = getTimeMs(); --currentDepth; break; }
        }
    }

    const auto& getLastFrame() const { return lastFrame; }
};

// 宏简化使用
#define PROFILE_SCOPE(name) \
    FrameProfiler::ScopeGuard _ps##__LINE__(g_profiler, name)

void gameUpdate(float dt) {
    PROFILE_SCOPE("GameUpdate");
    { PROFILE_SCOPE("Physics"); physics.step(); }
    { PROFILE_SCOPE("AI");      ai.update(dt); }
    { PROFILE_SCOPE("Render");  renderer.draw(); }
}
}

\note{**火焰图（Flame Graph）** 是可视化 profiling 数据的最佳方式：X 轴是时间，Y 轴是调用深度，每个色块宽度正比于耗时。一眼就能看到当前帧的性能瓶颈在哪个子系统。Chrome 的 chrome://tracing 可以加载 JSON 格式的 profiling 数据（Trace Event Format）。}

// 内存调试

\table{
工具 | 平台 | 检测能力
---

Valgrind (Memcheck) | Linux | 内存泄漏、未初始化读取

}

\code[cpp]{
// 自定义内存追踪器——重载全局 new/delete
#ifdef DEBUG_MEMORY
struct AllocRecord {
    const char* file;
    int line;
    size_t size;
};
static std::unordered_map<void*, AllocRecord> g_allocMap;
static std::mutex g_allocMutex;

void* operator new(size_t size, const char* file, int line) {
    void* ptr = std::malloc(size);
    std::lock_guard lock(g_allocMutex);
    g_allocMap[ptr] = {file, line, size};
    return ptr;
}
void operator delete(void* ptr) noexcept {
    std::lock_guard lock(g_allocMutex);
    g_allocMap.erase(ptr);
    std::free(ptr);
}

// 程序退出时打印未释放的内存
void reportLeaks() {
    for (auto& [ptr, rec] : g_allocMap) {
        std::cerr << "LEAK: " << rec.size << " bytes at "
                  << rec.file << ":" << rec.line << "\n";
    }
}
#define new new(__FILE__, __LINE__)
#endif
}

\warning{**调试代码必须能被完全排除。** 发布构建中 `#ifdef NDEBUG` 排除所有调试代码——Debug Draw、ImGui、Profiler 宏应编译为空操作。否则调试工具本身也会拖慢性能。}

// 日志系统

\code[cpp]{
// 分级日志系统
enum class LogLevel : uint8_t { Trace, Debug, Info, Warn, Error, Fatal };

class Logger {
    LogLevel minLevel = LogLevel::Info;
    std::vector<std::function<void(LogLevel, std::string_view)>> sinks;

public:
    void addSink(auto sink) { sinks.push_back(std::move(sink)); }

    void log(LogLevel level, std::string_view msg) {
        if (level < minLevel) return;
        for (auto& sink : sinks) sink(level, msg);
    }
};

// Sink 实现示例
auto consoleSink = [](LogLevel lv, std::string_view msg) {
    const char* prefix[] = {"[TRACE]","[DEBUG]","[INFO]","[WARN]","[ERROR]","[FATAL]"};
    std::cerr << prefix[int(lv)] << " " << msg << "\n";
};
auto fileSink = [&logFile](LogLevel, std::string_view msg) {
    logFile << msg << "\n";
};

// 使用宏添加文件/行号信息
#define LOG_INFO(fmt, ...) \
    g_logger.log(LogLevel::Info, std::format("[{}:{}] " fmt, __FILE__, __LINE__, ##__VA_ARGS__))
}

\tip{**日志设计最佳实践：** 1) 分级过滤——发布版只保留 Warn 以上；2) 分类标签（[Physics] [Audio] [Net]）方便 grep；3) 环形缓冲保留最近 N 条日志，崩溃时 dump 到文件；4) 线程安全的 sink，支持异步写入。}

// 命令行参数处理

\run[cpp]{
#include <iostream>
#include <string>
#include <unordered_map>
#include <vector>

class CmdLineParser {
    std::unordered_map<std::string, std::string> args;
    std::vector<std::string> positional;
public:
    void parse(int argc, const char* argv[]) {
        for (int i = 1; i < argc; ++i) {
            std::string arg = argv[i];
            if (arg.starts_with("--")) {
                auto eq = arg.find('=');
                if (eq != std::string::npos)
                    args[arg.substr(2, eq-2)] = arg.substr(eq+1);
                else if (i+1 < argc && argv[i+1][0] != '-')
                    args[arg.substr(2)] = argv[++i];
                else
                    args[arg.substr(2)] = "true";
            } else if (arg.starts_with("-")) {
                args[arg.substr(1)] = (i+1 < argc) ? argv[++i] : "true";
            } else {
                positional.push_back(arg);
            }
        }
    }
    std::string get(const std::string& key, const std::string& def = "") const {
        auto it = args.find(key);
        return it != args.end() ? it->second : def;
    }
    bool has(const std::string& key) const { return args.contains(key); }
};

int main() {
    // 模拟命令行参数
    const char* argv[] = {"game", "--width=1920", "--height", "1080",
                          "--fullscreen", "-v", "data/level1.scene"};
    int argc = 7;

    CmdLineParser cmd;
    cmd.parse(argc, argv);

    std::cout << "Width: " << cmd.get("width", "800") << "\n";
    std::cout << "Height: " << cmd.get("height", "600") << "\n";
    std::cout << "Fullscreen: " << cmd.get("fullscreen") << "\n";
    std::cout << "Verbose: " << cmd.get("v") << "\n";
}
}

/// 工程实践

\list{
- ImGui 面板用 `#ifdef EDITOR` 条件编译，发布版完全排除
- Profiler 数据导出为 Chrome Tracing JSON 格式，用浏览器分析
- ASan 默认在 Debug 构建中开启（CMake: `-fsanitize=address`）
- 日志系统在崩溃信号处理器（signal handler）中 flush 缓冲区
- 资产管道工具用 Python/C++ 命令行实现，集成到 CMake 构建流程
- Undo/Redo 的 Command 对象要可序列化——支持"撤销历史"保存和恢复
}

/// 难点易错点分析

\danger{**Debug 工具本身的性能开销。** 全局 `new`/`delete` 重载 + mutex 会让 Debug 构建慢 10~50 倍。Profile 宏在 Release 中必须编译为空。ImGui 的渲染开销在移动设备上也可能显著——只在开发机上启用。}

\warning{**日志格式化中的性能陷阱。** `std::format` 或 `printf` 在高频调用时开销不可忽视。用日志级别过滤——如果当前级别低于阈值，连格式化都不做（先检查级别再拼字符串）。}

\quiz[choice]{
Q: Dear ImGui 采用即时模式（Immediate Mode）的核心好处是？
A: 渲染速度比保留模式（Retained Mode）更快
B: 无需维护 Widget 状态树，代码逻辑与 UI 同步无需手动同步
C: 支持更复杂的动画效果
D: 可以直接输出 HTML
ANS: B
EXP: 即时模式每帧重建 UI——当前帧的 UI 总是反映当前的数据状态，不存在数据更新但 UI 没更新的同步问题。保留模式需要在数据变化时手动更新 Widget，容易导致状态不一致。代价是每帧都要做 UI 遍历，但对调试工具来说这个开销完全可接受。
}

\quiz[tf]{
Q: AddressSanitizer (ASan) 适合在发布版构建中始终启用。
ANS: F
EXP: ASan 会使程序变慢 2~5 倍且增加内存占用约 3 倍。它应该在 Debug/CI 构建中启用来检测内存错误，但发布版必须关闭。发布版可用更轻量的运行时检查（如自定义内存追踪器的 Release 简化版本）。
}

\quiz[choice]{
Q: 火焰图（Flame Graph）的 X 轴表示什么？
A: 任意排布，无固定含义
B: 函数调用的频率
C: 时间线（每个色块的宽度正比于该 scope 的耗时）
D: 内存使用量
ANS: C
EXP: 火焰图的 X 轴是时间，色块宽度正比于 scope 的持续时间，Y 轴是调用栈深度。宽色块就是耗时长的热点——一眼就能定位瓶颈。Chrome 的 chrome://tracing 工具可以直接加载 JSON 格式的 Trace Event 数据。
}

// 🟢 基础：帧性能分析器

\run[cpp]{
#include <iostream>
#include <vector>
#include <string>
#include <chrono>

class SimpleProfiler {
    struct Record { std::string name; double ms; };
    std::vector<Record> records;
    std::chrono::high_resolution_clock::time_point scopeStart;
    std::string currentScope;
public:
    void begin(const std::string& name) {
        currentScope = name;
        scopeStart = std::chrono::high_resolution_clock::now();
    }
    void end() {
        auto now = std::chrono::high_resolution_clock::now();
        double ms = std::chrono::duration<double, std::milli>(now - scopeStart).count();
        records.push_back({currentScope, ms});
    }
    void report() {
        std::cout << "=== Frame Profile ===\n";
        double total = 0;
        for (auto& r : records) {
            std::cout << "  " << r.name << ": " << r.ms << " ms\n";
            total += r.ms;
        }
        std::cout << "  TOTAL: " << total << " ms\n";
        records.clear();
    }
};

// TODO: 改进为 RAII ScopeGuard 风格，支持嵌套
int main() {
    SimpleProfiler prof;

    prof.begin("Physics");
    volatile int sum = 0;
    for (int i = 0; i < 1000000; ++i) sum += i;
    prof.end();

    prof.begin("AI");
    for (int i = 0; i < 500000; ++i) sum += i;
    prof.end();

    prof.report();
}
}
